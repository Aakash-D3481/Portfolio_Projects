CREATE TABLE Constituencies (
    State_ID INT,
    Constituency_Number INT,
    Constituency_Name VARCHAR(255),
    PRIMARY KEY (State_ID, Constituency_Number)
);

CREATE TABLE States (
    State_ID INT,
    State_Name VARCHAR(255),
    PRIMARY KEY (State_ID)
);

CREATE TABLE Years (
    Year_ID INT,
    Election_Year INT,
    PRIMARY KEY (Year_ID)
);

CREATE TABLE Election_Type (
    Election_Type_ID INT,
    Election_Name VARCHAR(255),
    PRIMARY KEY (Election_Type_ID)
);

CREATE TABLE Votes (
    Election_Year_ID INT,
    Election_Type_ID INT,
    State_ID INT,
    Constituency_Number INT,
	SI_No INT,
	PS_No VARCHAR(255),
    Parties VARCHAR(255),
    Votes INT,
	Total_Valid_Votes INT,
	NOTA INT,
	Total_Votes INT,
    FOREIGN KEY (Election_Year_ID) REFERENCES Years(Year_ID),
    FOREIGN KEY (Election_Type_ID) REFERENCES Election_Type(Election_Type_ID),
    FOREIGN KEY (State_ID, Constituency_Number) REFERENCES Constituencies(State_ID, Constituency_Number)
);
